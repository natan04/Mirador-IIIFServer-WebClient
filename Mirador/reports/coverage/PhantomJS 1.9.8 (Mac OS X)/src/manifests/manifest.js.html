      if (canvas.hasOwnProperty('thumbnail')) {

            version = $.Iiif.getVersionFromContext(service['@context']);

          thumbnailUrl = canvas.thumbnail['@id'];
        }

        var resource = canvas.images[0].resource;

          return resource['@id'];
        }

        if (service.hasOwnProperty('@context')) {
          version = $.Iiif.getVersionFromContext(service['@context']);
        }
